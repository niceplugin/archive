The 'myS3' slider enabled the background property.

In a demo page, content and background are the same color, so they can be confusing.

When customizing, you can show brilliant performance by applying background image etc. to background <li> tag.

<ul id="myS3">
    <li></li>
    <li></li>
    <li></li>
</ul>
<script>
    NP.Slider.set(
        {
            id: 'myS3',
            width: 640,
            height: 320
            code: {
                duration: 1000,
                bgAnimation: 0,
                background: 1
            }
        }
    );
</script>

The 'myS9' slider uses the YouTube Player API.

See YouTube Player API. link

Though not implemented on the demo site, you can visually disable the slider control buttons when playing a movie, depending on your abilities.

<ul id="myS9">
    <li>
        <div id="YTdemo"></div>
    </li>
    <li></li>
    <li></li>
</ul>
<script>
    NP.Slider.set(
        {
            id: 'myS9',
            width: 640,
            height: 480,
            event: myFunc2,
            code: {
                autoplay: 0
            }
        }
    );
    function myFunc2(a, b, c) {
        if (a === 'myS9' && b === 0 && myYTplayer.getPlayerState() === 1) {
            myYTplayer.pauseVideo();
        }
    }

    //About YouTube
    var myYTplayer,
        tag = document.createElement('script'),
        firstScriptTag = document.getElementsByTagName('script')[0];

    tag.src = "https://www.youtube.com/iframe_api";
    firstScriptTag.parentNode.insertBefore(tag, firstScriptTag);

    function onYouTubeIframeAPIReady() {
      myYTplayer = new YT.Player('YTdemo', {
        height: '100%',
        width: '640',
        videoId: '9g2U12SsRns'
      });
    }
</script>
