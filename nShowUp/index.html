This is a simple example for the implementation of 'parallax'.

NP.ShowUp.set(
    {
        content: '.parallax_box',
        add: 'noneAni',
        run: 'always',
        event: parallax_box,
        realTime: true
    }
);
function parallax_box(o, b, s) {
    var bg1 = document.getElementById('bg1'),
        bg2 = document.getElementById('bg2'),

        //The Y coordinate of the center content .
        m = (b.bottom - b.top) / 2,
        
        //The distance to move.
        //The number 4 indicates the level of movement.
        x = (b.top - m) / 4;

    bg1.style.top = x + 'px';
    bg2.style.top = -x + 'px';
}
